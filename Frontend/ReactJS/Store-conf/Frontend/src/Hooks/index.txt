<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Productos</title>
</head>
<body>
    <h1>Lista de Productos</h1>
    <ul id="productList"></ul>

    <script>
        function consultarAPI() {
            fetch('http://localhost:1337/api/products')
                .then(response => response.json())
                .then(data => {
                    console.log('Respuesta de la API:');
                    console.log(data);

                    // Obtener la lista de productos en el objeto 'data'
                    const productList = data.data;

                    // Obtener el elemento UL donde se mostrarán los productos
                    const ul = document.getElementById('productList');

                    // Usar map para crear una matriz de elementos <li>
                    const liElements = productList.map(product => {
                        const li = document.createElement('li');
                        li.innerHTML = `
                            <strong>ID:</strong> ${product.id}<br>
                            <strong>Título:</strong> ${product.attributes.title}<br>
                            <strong>Precio:</strong> $${product.attributes.price}<br>
                            <strong>Descripción:</strong> ${product.attributes.description}<br>
                            <strong>Fecha de Creación:</strong> ${product.attributes.createdAt}<br>
                            <br>
                        `;
                        return li;
                    });

                    // Agregar los elementos <li> al UL
                    liElements.forEach(li => {
                        ul.appendChild(li);
                    });
                })
                .catch(error => {
                    console.error('Error al consultar la API:', error);
                });
        }

        // Llama a la función para consultar la API
        consultarAPI();
    </script>
</body>
</html>
